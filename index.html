EasyNumber plugin examples

Example 1

Input with default options.

$('#js-easynumber-1').easyNumber();

Example 2

Input with button to increment or decrement customize by user .

$('#js-easynumber-2').easyNumber(){
plus: 'plus',
minus: 'minus',
});

Example 3

Input with readonly option.

$('#js-easynumber-3').easyNumber(){
readonly: true,
allowNegative: false
});

Example 4

Input without readonly option.

$('#js-easynumber-4').easyNumber(){
readonly: false,
});

Example 5

Input disallow to set negativ value.

$('#js-easynumber-5').easyNumber(){
readonly: true,
allowNegative: false
});

Example 6

Input allow to set negativ value.

$('#js-easynumber-6').easyNumber(){
allowNegative: false
});

Example 7

Input type option data.

$('#js-easynumber-7').easyNumber(){
allowNegative: true
});

Example 8

Input with option time.

$('#js-easynumber-8').easyNumber(){
type: 'day'
});

Example 9

Input with disable validation option.

$('#js-easynumber-9').easyNumber(){
type: 'time'
});

Example 10

Input with enable validation option.

$('#js-easynumber-10').easyNumber(){
validation: false
});

Example 11

Input wchich start value is another number that 0 (example 40).

$(('#js-easynumber-11').easyNumber(){
readonly: false ,
validation: true
});
